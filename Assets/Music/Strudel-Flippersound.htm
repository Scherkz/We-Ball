setcps(140/60/4)
register('acidenv', (x,pat) => pat.lpf(100)
.lpenv(x*9).lps(.2).lpd(.12)
)
const bassIntro =
stack(
n("<0 4 0 9 7>*16")
.scale("e:minor")
.trans(-9)
.s("sawtooth")
.acidenv(slider(0.526)),
n("<0>*16")
.scale("e:minor")
.trans(-18)
.detune(rand)
.s("supersaw")
.acidenv(slider(0.475))
)

const bassFull =
stack(
n("<0 4 0 9 7>*16")
.scale("e:minor")
.trans(-9)
.sometimesBy(0.2, x =>
x.scaleTranspose("<0 1 0 -1 0>".segment(-3))
.sometimesBy(0.2, y => y.transpose(-3))
)
.s("sawtooth")
.acidenv(slider(0.678)),
n("<0>*16")
.scale("e:minor")
.trans(-18)
.detune(rand)
.s("supersaw")
.acidenv(slider(0.678))
)

const bassTheme =
arrange(
[4, bassIntro],
[999, bassFull]
)

const drums =
stack(
s("top:1/2").fit().gain(0.3),
s("bd:2!4")
.duck("2:3:4:5")
.duckdepth(.8)
.duckattack(.16)
.gain(0.3)
)
const drumsTheme =
arrange(
[4, silence],
[999, drums]
)

$: stack(
bassTheme,
drumsTheme
)
._pianoroll()
